# SWEA 12004. 구구단1

```python
T = int(input())
for tc in range(1, T+1):
    N = int(input())
    num_list = [9, 8, 7, 6, 5, 4, 3, 2]
    ans_list = []

    for i in num_list:
        while N % i == 0:
            ans_list.append(i)
            N /= i



    if len(ans_list) == 0:
        ans = 'No'
    elif len(ans_list) <= 2:
        ans = 'Yes'
    else:
        ans = 'No'

    print('#{} {}'.format(tc, ans))
```
