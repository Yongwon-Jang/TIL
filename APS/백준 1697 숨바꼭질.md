```python
def bfs(n):
    global min_cnt
    Q.append(n)
    while Q:
        x = Q.pop(0)
        if x == K:
            min_cnt = visited[x]
            break
        for i in range(3):
            if i == 0: nx = x-1
            elif i == 1: nx = x+1
            elif i == 2: nx = x*2
            if 0 <= nx <= lim and not visited[nx]:
                visited[nx] = visited[x] + 1
                Q.append(nx)


N, K = map(int, input().split())

Q = []
lim = 100000
visited = [0] * (lim+1)
min_cnt = 987654321
bfs(N)

print(min_cnt)
```
