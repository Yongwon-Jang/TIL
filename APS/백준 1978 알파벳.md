def dfs(y, x, cnt):
    global max_cnt
    # if cnt > max_cnt:
    #     max_cnt = cnt
    max_cnt = max(cnt, max_cnt)

    for i in range(4):
        nx = x + dx[i]
        ny = y + dy[i]
        if 0 <= nx < C and 0 <= ny < R and arr[ny][nx] not in stack:
            stack.append(arr[ny][nx])
            dfs(ny, nx, cnt+1)
            stack.pop(-1)

R, C = map(int, input().split())
# arr = [list(input()) for _ in range(R)]
arr = [list(map(lambda a : ord(a)-65,input())) for _ in range(R)]

dx = [-1, 0, 1, 0]
dy = [0, -1, 0, 1]
max_cnt = 0
stack = [arr[0][0]]
