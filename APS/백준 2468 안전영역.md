```python
def bfs(y, x):
    rain_check[y][x] = 1
    Q.append([y, x])
    while Q:
        t = Q.pop(0)
        for i in range(4):
            nx = t[1] + dx[i]
            ny = t[0] + dy[i]
            if 0 <= nx < N and 0 <= ny < N:
                if rain_check[ny][nx] == 0:
                    rain_check[ny][nx] = 1
                    Q.append([ny, nx])


N = int(input())
arr = [list(map(int, input().split())) for _ in range(N)]

dx = [0, -1, 0, 1]
dy = [-1, 0, 1, 0]

max_rain = 0
max_cnt = 0

for i in range(N):
    for j in range(N):
        if arr[i][j] > max_rain:
            max_rain = arr[i][j]

for i in range(max_rain):
    rain_check = [[0]*N for _ in range(N)]
    for j in range(N):
        for k in range(N):
            if arr[j][k] <= i:
                rain_check[j][k] = 1
    cnt = 0
    Q = []

    for j in range(N):
        for k in range(N):
            if rain_check[j][k] == 0:
                cnt += 1
                bfs(j, k)
    if cnt > max_cnt:
        max_cnt = cnt
print(max_cnt)
```
