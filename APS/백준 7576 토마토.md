```python
import sys
sys.stdin = open('토마토_input.txt')

from collections import deque

def bfs():
    while Q:
        t = Q.popleft()
        for i in range(4):
            nx = t[1] + dx[i]
            ny = t[0] + dy[i]
            if 0 <= nx < M and 0 <= ny < N and arr[ny][nx] == 0:
                arr[ny][nx] = arr[t[0]][t[1]] + 1
                Q.append([ny, nx])

def max_check():
    global max_ans
    for i in range(N):
        for j in range(M):
            if arr[i][j] > max_ans:
                max_ans = arr[i][j]
            if arr[i][j] == 0:
                max_ans = 0
                return max_ans
    return max_ans


M, N = map(int, input().split())
arr = [list(map(int, input().split())) for _ in range(N)]

Q = deque()
dx = [-1, 0, 1, 0]
dy = [0, -1, 0, 1]

for i in range(N):
    for j in range(M):
        if arr[i][j] == 1:
            Q.append([i, j])

bfs()
# print(arr)
max_ans = 0
max_check()

print(max_ans-1)
```
