#  프로그래머스(모의고사)

```python
def solution(answers):
    N = len(answers)
    A = []
    B = []
    C = []
    B_dict = {1 : 1, 3: 3, 5: 4, 7: 5}
    C_list = [3, 3, 1, 1, 2, 2, 4, 4, 5, 5]
    for i in range(N):
        A.append((i%5)+1)
        if i%2 == 0:
            B.append(2)
        else:
            B.append(B_dict[i%8])
        C.append(C_list[i%10])
    A_cnt = 0
    B_cnt = 0
    C_cnt = 0
    for i in range(N):
        if answers[i] == A[i]:
            A_cnt += 1
        if answers[i] == B[i]:
            B_cnt += 1
        if answers[i] == C[i]:
            C_cnt += 1

    ans_list = [A_cnt, B_cnt, C_cnt]
    max_ans = max(ans_list)
    answer = []
    for i in range(len(ans_list)):
        if ans_list[i] == max_ans:
            answer.append(i+1)

    return answer
```
