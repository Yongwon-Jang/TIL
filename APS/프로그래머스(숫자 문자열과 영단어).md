# 프로그래머스(숫자 문자열과 영단어)

```python
def solution(s):
    num = {'zero':'0', 'one':'1', 'two':'2', 'three':'3',
           'four':'4', 'five':'5', 'six':'6', 'seven':'7',
           'eight':'8', 'nine':'9'}
    num_add = ''
    answer = []
    for i in s:
        num_add = num_add + i
        if '0' <= num_add <= '9':
            answer.append(num_add)
            num_add = ''
        else:
            if num_add in num:
                answer.append(num[num_add])
                num_add = ''
    return int(''.join(answer))

print(solution("2three45sixseven"))
```
