# 프로그래머스(프렌즈4블록)

```python
import copy

def solution(m, n, board):
    def rec(y, x):
        a = board[y][x]
        cnt = 0
        for i in range(y, y+2):
            for j in range(x, x+2):
                if board[i][j] == a and a != -1:
                    cnt += 1
                else:
                    return
        check.append([y, x])
    for i in range(m):
        board[i] = list(board[i])

    check = []
    for i in range(m-1):
        for j in range(n-1):
            rec(i, j)
    if check:
        for i in check:
            for j in range(i[0], i[0]+2):
                for k in range(i[1], i[1]+2):
                    board[j][k] = -1
        while True:
            check_board = copy.deepcopy(board)
            for i in range(m-1):
                for j in range(n):
                    if board[i+1][j] == -1:
                        board[i][j], board[i+1][j] = board[i+1][j], board[i][j]
            if board == check_board:
                break

        solution(m, n, board)
    answer = 0
    for i in range(m):
        for j in range(n):
            if board[i][j] == -1:
                answer += 1
    return answer

print(solution(4, 5, ["CCBDE", "AAADE", "AAABF", "CCBBF"]))
```
